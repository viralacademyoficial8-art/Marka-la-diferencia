Mejora la precisión del tracking con eventos del servidor

💡
¿Por qué usar Conversions API?

Con iOS 14.5+ y bloqueadores de anuncios, el Pixel pierde hasta 30% de las conversiones. CAPI envía eventos directamente desde tu servidor a Meta, mejorando la atribución.

Activar Conversions API Enviar eventos duplicados al servidor de Meta
Genera un token en: Meta Business Suite → Configuración → Conversions API
Usa esto para probar eventos antes de ir a producción

// 1. Cuando abre el modal de registro
function trackInitiateRegistration() {
  fbq('track', 'InitiateCheckout', {
    content_name: 'Masterclass Registration',
    content_category: 'Lead Form'
  });
}

// 2. Cuando envía el formulario (LEAD)
function trackLead(userData) {
  fbq('track', 'Lead', {
    content_name: 'Masterclass No Es Tu Producto',
    content_category: 'Webinar Registration',
    value: 5.00,
    currency: 'USD'
  }, {
    eventID: 'lead_' + Date.now() // Para deduplicación
  });
}

// 3. Cuando llega a página de confirmación
function trackCompleteRegistration() {
  fbq('track', 'CompleteRegistration', {
    content_name: 'Masterclass Confirmed',
    status: 'registered'
  });
}

// 4. Cuando hace clic en WhatsApp
function trackWhatsAppJoin() {
  fbq('trackCustom', 'WhatsAppJoin', {
    content_name: 'WhatsApp Group',
    funnel_step: 'post_registration'
  });
}

// 5. Cuando agrega al calendario
function trackSchedule() {
  fbq('track', 'Schedule', {
    content_name: 'Masterclass Calendar',
    calendar_type: 'google' // o 'apple', 'outlook'
  });
}

// 6. Cuando reproduce el video
function trackVideoView(percent) {
  fbq('track', 'ViewContent', {
    content_name: 'VSL Video',
    content_type: 'video',
    video_percent: percent
  });
}

// 7. Cuando hace clic en VIP
function trackVIPIntent() {
  fbq('track', 'InitiateCheckout', {
    content_name: 'VIP Access',
    value: 14.00,
    currency: 'USD'
  });
}

// 8. Cuando compra VIP (conectar con pasarela)
function trackPurchase(orderId) {
  fbq('track', 'Purchase', {
    content_name: 'VIP Access',
    content_type: 'product',
    value: 14.00,
    currency: 'USD',
    order_id: orderId
  });
}
📄 conversions-api.js (Server-side / Supabase Edge Function) 
// Supabase Edge Function para Conversions API
import { serve } from 'https://deno.land/std@0.168.0/http/server.ts'
import { createHash } from 'https://deno.land/std@0.168.0/crypto/mod.ts'

const PIXEL_ID = Deno.env.get('META_PIXEL_ID')
const ACCESS_TOKEN = Deno.env.get('META_ACCESS_TOKEN')

// Función para hashear datos (SHA256)
function hashData(data: string): string {
  const encoder = new TextEncoder()
  const hash = createHash('sha256')
  hash.update(encoder.encode(data.toLowerCase().trim()))
  return hash.toString()
}

serve(async (req) => {
  const { event_name, user_data, custom_data, event_id } = await req.json()
  
  const payload = {
    data: [{
      event_name,
      event_time: Math.floor(Date.now() / 1000),
      event_id, // Para deduplicación con Pixel
      action_source: 'website',
      user_data: {
        em: user_data.email ? [hashData(user_data.email)] : undefined,
        ph: user_data.phone ? [hashData(user_data.phone)] : undefined,
        fn: user_data.firstName ? [hashData(user_data.firstName)] : undefined,
        client_ip_address: req.headers.get('x-forwarded-for'),
        client_user_agent: req.headers.get('user-agent'),
        fbc: user_data.fbc, // Facebook Click ID
        fbp: user_data.fbp, // Facebook Browser ID
      },
      custom_data
    }]
  }

  const response = await fetch(
    `https://graph.facebook.com/v18.0/${PIXEL_ID}/events?access_token=${ACCESS_TOKEN}`,
    {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(payload)
    }
  )

  return new Response(JSON.stringify(await response.json()), {
    headers: { 'Content-Type': 'application/json' }
  })
})
